```scheme
(define (exp a b)
  (define (iter count result)
    (if (= 1 count)
        result
        (iter (- count 1) (* a result))))
  
  (if (= 0 b)
      1
      (iter b a)))

(define (cons x y)
  (* (exp 2 x) (exp 3 y)))

(define (car z)
  (cond ((not (= 0 (remainder n 2))) 0)
        (else (+ (car (/ n 2) 2) 1))))

(define (cdr z)
  (cond ((not (= 0 (remainder n 3))) 0)
        (else (+ (car (/ n 3) 3) 1))))
```
