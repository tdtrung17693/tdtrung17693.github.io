```scheme
(define (last-pair l)
  (if (null? (cdr l))
      l
      (last-pair (cdr l))))

(define (append list1 list2)
  (if (null? list1)
      list2
      (cons (car list1) (append (cdr list1) list2))))

(define (length items)
  (if (null? items)
      0
      (+ 1 (length (cdr items)))))

(define (reverse items)
  (if (null? items)
      items
      (append (reverse (cdr items)) (list (car items)))))

(reverse (list 1 2 3 4))
```
